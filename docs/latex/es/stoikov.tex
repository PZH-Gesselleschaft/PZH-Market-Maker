\section{Anwendung der Formel}

Den fairen Preis anhand der Formel $Mikropreis = M + G$ schätzen

\section{Aufschlüsselung der Grundformel}

\subsection{Terminología}

$bid-price = bid^p$

$ask-price = ask^p$

$bid-size  = bid^s$

$bid-size  = ask^s$

\subsection{Fórmula de Micro-precio}

$M = \frac{bid^p + ask^p}{2}$

$S = ask^p - bid^p$

$I = \frac{bid^s}{bid^s + ask^s}$

$G = \left(I - 0.5\right) \cdot S$

$Mikropreis = M + G$

\subsection{Fórmula de Avellaneda–Stoikov}
